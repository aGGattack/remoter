The following deliverables are expected from this project:

\subsection*{Hardware Deliverables}
\begin{enumerate}
    \item \textbf{Remote Controller Unit}
    \begin{itemize}
        \item Custom PCB with microcontroller, radio transceiver, joystick controls, and OLED display
        \item 3D-printed enclosure
        \item Battery management system (8+ hour runtime)
    \end{itemize}
    
    \item \textbf{Robot Receiver Unit}
    \begin{itemize}
        \item Custom PCB with microcontroller, radio transceiver, and motor control interface
        \item 3D-printed enclosure
        \item Power management for robot-side operation
    \end{itemize}
    
    \item \textbf{Circuit Diagrams and PCB Layouts}
    \begin{itemize}
        \item Complete schematics for both units
        \item Gerber files for PCB manufacturing
        \item Bill of materials (BOM)
    \end{itemize}
\end{enumerate}

\subsection*{Software Deliverables}
\begin{enumerate}
    \item \textbf{Firmware for Remote Unit}
    \begin{itemize}
        \item Joystick input processing
        \item Radio communication protocol
        \item OLED display driver
        \item Battery monitoring
    \end{itemize}
    
    \item \textbf{Firmware for Robot Unit}
    \begin{itemize}
        \item Radio communication handler
        \item Motor control interface
        \item Communication loss detection
        \item Safety shutdown logic
    \end{itemize}
\end{enumerate}

\subsection*{Documentation Deliverables}
\begin{enumerate}
    \item \textbf{Technical Report} (Bachelor thesis, approx. 40-60 pages)
    \item \textbf{User Manual} describing operation of the system
    \item \textbf{Test Report} documenting validation results
    \item \textbf{Source Code} (documented and version-controlled)
\end{enumerate}
